zabbix端口监控脚本

创建脚本check_port_add.sh

#!/bin/bash 

#判断脚本是否存在
#判断端口是否已经存在，如果存在则不添加，不存在的话添加

ZABBIX_CONF_FILE=/etc/zabbix/zabbix_agentd.d/externalscripts/check_port.py
TCP_PORT=$1
SERVICE_NAME=$2

echo "TCP端口号是：$TCP_PORT"
echo "服务名是：$SERVICE_NAME"


if [ ! -e $ZABBIX_CONF_FILE ] ;then
   echo "$ZABBIX_CONF_FILE 不存在"
   exit 1
fi

grep $TCP_PORT $ZABBIX_CONF_FILE
if [ $? -eq 0 ] ;then
   echo "端口已存在，不需要重复添加"
   exit 1
fi

#添加端口

sed -i "/total_dict=/ a\{\"{#TCP_PORT}\":\"$TCP_PORT\",\"{#SERVICE_NAME}\":\"$SERVICE_NAME\"}," $ZABBIX_CONF_FILE

#输出total_dict后三行
grep total_dict -A 3 $ZABBIX_CONF_FILE
#在total_dict后面插入一行，$TCP_PORT和$SERVICE_NAME是变量，所以sed需要用双引号,因为双引号是弱引用，会把变量转换成值再进行替换，单引号是强引用

#重启zabbix
echo "重启zabbix"
echo "systemctl restart zabbix-agent"
#或许不需要重启zabbix-agent
#systemctl restart zabbix-agent

怎么使用

配合 /root/deploy/deploy.sh 部署脚本使用，在部署脚本最后添加函数

#函数
zabbix(){
   bash /opt/script/check_port_add.sh $server_port $prj_script
}

#执行
zabbix
